\section{Logical Or $\lor$}
\subsection{Formal Definition}
\begin{center}
$
\lor = \lor \lbrack{a_i,a_j}\rbrack \equiv 
$
\end{center}
\begin{center}
$
\exists a_x \in \{a_i,a_j\} : a_x == True, \hspace{3mm} \lor \lbrack a_i,a_j \rbrack = True
$
\end{center}
\begin{center}
$
\not \exists a_x \in \{a_i,a_j\} : a_x == True, \hspace{3mm} \lor \lbrack a_i,a_j \rbrack = False
$
\end{center}
\subsection{Alternate Definition}
$\lor$ \lbrack{False, False}\rbrack  \hspace{.8mm} = False \\
 $\lor$ \lbrack{False, True}\rbrack  \hspace{.8mm} = True \\
 $\lor$ \lbrack{True, False}\rbrack \hspace{.8mm} = True \\
 $\lor$ \lbrack{True,True}\rbrack   \hspace{.8mm} = True

\section{Logical And $\land$}
\subsection{Definition}
\begin{center}
$
\land = \land \lbrack{a_i,a_j}\rbrack \equiv 
$
\end{center}
\begin{center}
$
\exists a_x \in \{a_i,a_j\} : a_x == False, \hspace{3mm} \land \lbrack a_i,a_j \rbrack = False
$
\end{center}
\begin{center}
$
\not \exists a_x \in \{a_i,a_j\} : a_x == False, \hspace{3mm} \land \lbrack a_i,a_j \rbrack = True
$
\end{center}
\subsection{Alternate Definition}
$\land$ \lbrack{False, False}\rbrack = False \\
 $\land$ \lbrack{False, True}\rbrack = False \\
 $\land$ \lbrack{True, False}\rbrack = False \\
 $\land$ \lbrack{True,True}\rbrack  = True

\section{Law of Non-Contradiction}
\begin{center}
Law of Non-Contradiction 
\end{center}

\subsection{Proof}
Assumptions\\
1. Let $a_i \rightarrow$ True, $\hspace{2mm} \forall i$
\\
Assertions\\
2. Suppose $\exists i : \neg a_i \rightarrow$ True\\
3. $a_i \land \neg a_i \rightarrow \perp$ \hspace{3mm} (implied from 2)\\
4. $\neg(\exists i : \neg a_i) \rightarrow$ True\\
\ \ \ $\not \exists i : \neg a_i \rightarrow True$\\
5. $\therefore a_i = a_i \hspace{3mm} \forall i$
